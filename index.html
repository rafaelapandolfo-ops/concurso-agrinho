LÓGICA DO SIMULADOR AGROAMBIENTAL // ========================================== // Estado inicial do simulador const estadoSimulador = { ilpf: { ativo: false, arvores: 450, co2: 12.5, prod: 20 }, biologicos: { ativo: false, arvores: 0, co2: 4.2, prod: 15 }, satelite: { ativo: false, arvores: 1200, co2: 35.0, prod: -5 } // Foco total em proteção ambiental }; function ativarTecnologia(idTecnologia) { const tecnologia = estadoSimulador[idTecnologia]; const botao = document.querySelector(`button[onclick="ativarTecnologia('${idTecnologia}')"]`); const logSistema = document.getElementById('log-sistema'); if (!tecnologia) return; // Alterna o estado ativo/inativo tecnologia.ativo = !tecnologia.ativo; // Atualiza a interface do botão (efeito visual de selecionado) if (tecnologia.ativo) { botao.classList.add('active'); // Você pode estilizar a classe .active no seu CSS botao.style.border = "2px solid #2ecc71"; botao.style.backgroundColor = "rgba(46, 204, 113, 0.1)"; adicionarLog(`[Sucesso]: Diretriz '${idTecnologia.toUpperCase()}' ativada com sucesso.`); } else { botao.classList.remove('active'); botao.style.border = ""; botao.style.backgroundColor = ""; adicionarLog(`[Aviso]: Diretriz '${idTecnologia.toUpperCase()}' desativada.`); } // Recalcula as métricas globais atualizarMetricas(); } function adicionarLog(mensagem) { const logSistema = document.getElementById('log-sistema'); const timestamp = new Date().toLocaleTimeString(); logSistema.innerHTML = `[${timestamp}] ${mensagem}`; } function atualizarMetricas() { let totalArvores = 0; let totalCO2 = 0.0; let totalProd = 100; // Produtividade base de 100% // Soma os impactos de cada tecnologia ativa for (const chave in estadoSimulador) { if (estadoSimulador[chave].ativo) { totalArvores += estadoSimulador[chave].arvores; totalCO2 += estadoSimulador[chave].co2; totalProd += estadoSimulador[chave].prod; } } // Atualiza os elementos na tela document.getElementById('m-arvores').innerText = totalArvores.toLocaleString('pt-BR'); document.getElementById('m-co2').innerText = `${totalCO2.toFixed(1)} t`; document.getElementById('m-prod').innerText = `${totalProd}%`; } // ========================================== // 2. LÓGICA DO QUIZ AGROSUSTENTÁVEL // ========================================== // Banco de perguntas focado no tema e nas fontes descritas no HTML const bancoPerguntas = [ { pergunta: "De acordo com o texto: "Áreas de Preservação Permanente (APPs) e Reserva Legal", correta: true }, { texto: "Zonas de Amortecimento Industriais", correta: false }, { texto: "Apenas propriedades latifundiárias de monocultura", correta: false } ] }, { pergunta: "Qual é a principal vantagem ambiental da tecnologia ILPF (Integração Lavoura-Pecuária-Floresta) promovida pela Embrapa?", respostas: [ { texto: "Maximizar o desmatamento para a entrada de maquinário pesado", correta: false }, { texto: "Otimizar o espaço e promover a fixação biológica de carbono, mitigando gases estufa", correta: true }, { texto: "Substituir completamente a produção vegetal por criação de gado de corte", correta: false }, { texto: "Eliminar a necessidade de rotação de culturas na propriedade", correta: false } ] }, { pergunta: "Como a Intensificação Tecnológica (ex: plantio direto na palha) ajuda a evitar a expansão do desmatamento?", respostas: [ { texto: "Ela força o agricultor a buscar novas terras a cada safra", correta: false }, { texto: "Ela permite obter mais safras no mesmo espaço, poupando a abertura de florestas nativas", correta: true }, { texto: "Ela reduz a produtividade, tornando a terra inutilizável", correta: false }, { texto: "Ela exige a compactação severa do solo para funcionar", correta: false } ] }, { pergunta: "Qual ferramenta geoespacial listada nas fontes oficiais ajuda a monitorar em tempo real dados de desmatamento e uso do solo no Brasil?", respostas: [ { texto: "MapBiomas", correta: true }, { texto: "Google Tradutor", correta: false }, { texto: "Fórum Econômico Mundial", correta: false }, { texto: "Censo Demográfico do IBGE", correta: false } ] } ]; let indicePerguntaAtual = 0; let pontuacao = 0; // Elementos do DOM do Quiz const questionContainer = document.getElementById('question-container'); const questionTextElement = document.getElementById('question-text'); const answerButtonsElement = document.getElementById('answer-buttons'); const quizControls = document.getElementById('quiz-controls'); const scoreContainer = document.getElementById('score-container'); const scoreTextElement = document.getElementById('score-text'); // Inicializa o quiz quando a página carregar window.addEventListener('DOMContentLoaded', () => { iniciarQuiz(); }); function iniciarQuiz() { indicePerguntaAtual = 0; pontuacao = 0; scoreContainer.classList.add('hide'); scoreContainer.style.display = 'none'; questionContainer.classList.remove('hide'); questionContainer.style.display = 'block'; definirProximaPergunta(); } function definirProximaPergunta() { resetarEstadoQuiz(); mostrarPergunta(bancoPerguntas[indicePerguntaAtual]); } function mostrarPergunta(perguntaObjeto) { questionTextElement.innerText = perguntaObjeto.pergunta; perguntaObjeto.respostas.forEach(resposta => { const botao = document.createElement('button'); botao.innerText = resposta.texto; botao.classList.add('btn-quiz-opt'); // Adicione estilo para os botões de opção no CSS // Estilos básicos padrão para os botões do quiz botao.style.padding = "12px"; botao.style.margin = "5px 0"; botao.style.border = "1px solid #ccc"; botao.style.borderRadius = "6px"; botao.style.cursor = "pointer"; botao.style.textAlign = "left"; botao.style.backgroundColor = "#fff"; botao.style.transition = "all 0.2s ease"; if (resposta.correta) { botao.dataset.correta = resposta.correta; } botao.addEventListener('click', selecionarResposta); answerButtonsElement.appendChild(botao); }); } function resetarEstadoQuiz() { quizControls.classList.add('hide'); quizControls.style.display = 'none'; while (answerButtonsElement.firstChild) { answerButtonsElement.removeChild(answerButtonsElement.firstChild); } } function selecionarResposta(e) { const botaoSelecionado = e.target; const ehCorreto = botaoSelecionado.dataset.correta === "true"; if (ehCorreto) { pontuacao++; botaoSelecionado.style.backgroundColor = "#2ecc71"; // Verde para correto botaoSelecionado.style.color = "#fff"; } else { botaoSelecionado.style.backgroundColor = "#e74c3c"; // Vermelho para errado botaoSelecionado.style.color = "#fff"; } // Revela a resposta correta para o usuário e desativa cliques adicionais Array.from(answerButtonsElement.children).forEach(botao => { if (botao.dataset.correta === "true") { botao.style.backgroundColor = "#2ecc71"; botao.style.color = "#fff"; } botao.disabled = true; }); // Gerencia o fluxo das perguntas if (bancoPerguntas.length > indicePerguntaAtual + 1) { quizControls.classList.remove('hide'); quizControls.style.display = 'block'; } else { // Fim do quiz após um breve delay para visualização do resultado setTimeout(mostrarResultadoFinal, 1500); } } function proximaPergunta() { indicePerguntaAtual++; definirProximaPergunta(); } function mostrarResultadoFinal() { questionContainer.classList.add('hide'); questionContainer.style.display = 'none'; quizControls.classList.add('hide'); quizControls.style.display = 'none'; scoreContainer.classList.remove('hide'); scoreContainer.style.display = 'block'; scoreTextElement.innerText = `Você acertou ${pontuacao} de ${bancoPerguntas.length} perguntas.`; } function reiniciarQuiz() { iniciarQuiz(); }
